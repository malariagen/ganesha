DROP TABLE IF EXISTS `sample_classifications1`;
CREATE TABLE `sample_classifications1` (
  `ordr` int(11),
  `name` varchar(100) DEFAULT NULL,
  `sample_classification` varchar(50) NOT NULL,
  `sample_classification_type` varchar(50) DEFAULT NULL,
  `lattit` double DEFAULT NULL,
  `longit` double DEFAULT NULL,
  `geo_json` text,
  PRIMARY KEY (`sample_classification`) 
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

        LINES TERMINATED BY '\n'
        IGNORE 1 LINES 
        (
		ordr,name,sample_classification,sample_classification_type,lattit,longit,geo_json
        );
